How to access postgresql:
In windows, navigate to "C:\Program Files\PostgreSQL\18\bin" (and in linux no worries - already in path)
then type:
psql -U postgres -d hobbyfinder


CREATE TABLE genres (
    id SERIAL PRIMARY KEY,
    name VARCHAR(50) UNIQUE NOT NULL
);

INSERT INTO genres (name) VALUES
('sci-fi'),
('fantasy'),
('mystery'),
('ambient'),
('adventure');

CREATE TABLE hobbies (
    id SERIAL PRIMARY KEY,
    name VARCHAR(50) UNIQUE NOT NULL
);

INSERT INTO hobbies (name) VALUES
('Reading'),
('Streaming'),
('Listening to Music');

CREATE TABLE catalog (
    id SERIAL PRIMARY KEY,
    title VARCHAR(150) NOT NULL,
    type VARCHAR(50) NOT NULL,
    genre_id INTEGER REFERENCES genres(id)
);

INSERT INTO catalog (title, type, genre_id) VALUES
('Interstellar', 'movie', 1),
('Blade Runner', 'movie', 1),
('Dune', 'book', 1),
('The Hobbit', 'book', 2),
('Sherlock Holmes', 'book', 3),
('Lo-Fi Beats', 'music', 4),
('Star Wars', 'movie', 5);

CREATE TABLE recommendations (
    id SERIAL PRIMARY KEY,
    user_id VARCHAR(100) NOT NULL,
    catalog_id INTEGER REFERENCES catalog(id),
    hobby_id INTEGER REFERENCES hobbies(id),
    recommended_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
